Data Table 
q <- df %>%
  dplyr::select(ferm_number, fermenter, strain, run_description, target_product) %>%
  na.omit() %>%
  dplyr::group_by(ferm_number, fermenter, strain, run_description) %>%
  summarise(maxTargetProduct = max(target_product)) %>%
  mutate(maxTargetProduct = as.numeric(format(maxTargetProduct, digits = 4))) %>%
  arrange(strain, run_description, fermenter)

q$maxTargetProductScaled = rescale(q$maxTargetProduct, to = c(3, 15))
q$maxTargetProductScaled = as.numeric(format(q$maxTargetProductScaled, digits = 4))

datatable(q)
Strain_Run 
zxc <- q %>%

ggplot(zxc, aes(x = strain_run, y = maxTargetProduct)) +
